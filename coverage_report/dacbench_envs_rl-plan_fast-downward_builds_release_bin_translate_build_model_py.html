13def convert_rules(prog): 

14 RULE_TYPES = { 

15 "join": JoinRule, 

16 "product": ProductRule, 

17 "project": ProjectRule, 

18 } 

19 result = [] 

20 for rule in prog.rules: 

21 RuleType = RULE_TYPES[rule.type] 

22 new_effect, new_conditions = variables_to_numbers( 

23 rule.effect, rule.conditions) 

24 rule = RuleType(new_effect, new_conditions) 

25 rule.validate() 

26 result.append(rule) 

27 return result

29def variables_to_numbers(effect, conditions): 

30 new_effect_args = list(effect.args) 

31 rename_map = {} 

32 for i, arg in enumerate(effect.args): 

33 if arg[0] == "?": 

34 rename_map[arg] = i 

35 new_effect_args[i] = i 

36 new_effect = pddl.Atom(effect.predicate, new_effect_args) 

37 

38 # There are three possibilities for arguments in conditions: 

39 # 1. They are variables that occur in the effect. In that case, 

40 # they are replaced by the corresponding position in the 

41 # effect, as indicated by the rename_map. 

42 # 2. They are constants. In that case, the unifier must guarantee 

43 # that they are matched appropriately. In that case, they are 

44 # not modified (remain strings denoting objects). 

45 # 3. They are variables that don't occur in the effect (are 

46 # projected away). This is only allowed in projection rules. 

47 # Such arguments are also not modified (remain "?x" strings). 

48 

49 new_conditions = [] 

50 for cond in conditions: 

51 new_cond_args = [rename_map.get(arg, arg) for arg in cond.args] 

52 new_conditions.append(pddl.Atom(cond.predicate, new_cond_args)) 

53 return new_effect, new_conditions

55class BuildRule: 

56 def prepare_effect(self, new_atom, cond_index): 

57 effect_args = list(self.effect.args) 

58 cond = self.conditions[cond_index] 

59 for var_no, obj in zip(cond.args, new_atom.args): 

60 if isinstance(var_no, int): 

61 effect_args[var_no] = obj 

62 return effect_args 

63 def __str__(self): 

64 return "%s :- %s" % (self.effect, ", ".join(map(str, self.conditions))) 

65 def __repr__(self): 

66 return "<%s %s>" % (self.__class__.__name__, self)

68class JoinRule(BuildRule): 

69 def __init__(self, effect, conditions): 

70 self.effect = effect 

71 self.conditions = conditions 

72 left_args = conditions[0].args 

73 right_args = conditions[1].args 

74 left_vars = set([var for var in left_args if isinstance(var, int)]) 

75 right_vars = set([var for var in right_args if isinstance(var, int)]) 

76 common_vars = sorted(left_vars & right_vars) 

77 self.common_var_positions = [ 

78 [args.index(var) for var in common_vars] 

79 for args in (list(left_args), list(right_args))] 

80 self.atoms_by_key = ({}, {}) 

81 def validate(self): 

82 assert len(self.conditions) == 2, self 

83 left_args = self.conditions[0].args 

84 right_args = self.conditions[1].args 

85 eff_args = self.effect.args 

86 left_vars = set([v for v in left_args 

87 if isinstance(v, int) or v[0] == "?"]) 

88 right_vars = set([v for v in right_args 

89 if isinstance(v, int) or v[0] == "?"]) 

90 eff_vars = set([v for v in eff_args 

91 if isinstance(v, int) or v[0] == "?"]) 

92 assert left_vars & right_vars, self 

93 assert (left_vars | right_vars) == (left_vars & right_vars) | eff_vars, self 

94 def update_index(self, new_atom, cond_index): 

95 ordered_common_args = [ 

96 new_atom.args[position] 

97 for position in self.common_var_positions[cond_index]] 

98 key = tuple(ordered_common_args) 

99 self.atoms_by_key[cond_index].setdefault(key, []).append(new_atom) 

100 def fire(self, new_atom, cond_index, enqueue_func): 

101 effect_args = self.prepare_effect(new_atom, cond_index) 

102 ordered_common_args = [ 

103 new_atom.args[position] 

104 for position in self.common_var_positions[cond_index]] 

105 key = tuple(ordered_common_args) 

106 other_cond_index = 1 - cond_index 

107 other_cond = self.conditions[other_cond_index] 

108 for atom in self.atoms_by_key[other_cond_index].get(key, []): 

109 for var_no, obj in zip(other_cond.args, atom.args): 

110 if isinstance(var_no, int): 

111 effect_args[var_no] = obj 

112 enqueue_func(self.effect.predicate, effect_args)

114class ProductRule(BuildRule): 

115 def __init__(self, effect, conditions): 

116 self.effect = effect 

117 self.conditions = conditions 

118 self.atoms_by_index = [[] for c in self.conditions] 

119 self.empty_atom_list_no = len(self.conditions) 

120 def validate(self): 

121 assert len(self.conditions) >= 2, self 

122 cond_vars = [set([v for v in cond.args 

123 if isinstance(v, int) or v[0] == "?"]) 

124 for cond in self.conditions] 

125 all_cond_vars = reduce(set.union, cond_vars) 

126 eff_vars = set([v for v in self.effect.args 

127 if isinstance(v, int) or v[0] == "?"]) 

128 assert len(all_cond_vars) == len(eff_vars), self 

129 assert len(all_cond_vars) == sum([len(c) for c in cond_vars]) 

130 def update_index(self, new_atom, cond_index): 

131 atom_list = self.atoms_by_index[cond_index] 

132 if not atom_list: 

133 self.empty_atom_list_no -= 1 

134 atom_list.append(new_atom) 

135 

136 def _get_bindings(self, atom, cond): 

137 return [(var_no, obj) for var_no, obj in zip(cond.args, atom.args) 

138 if isinstance(var_no, int)] 

139 

140 def fire(self, new_atom, cond_index, enqueue_func): 

141 if self.empty_atom_list_no: 

142 return 

143 

144 # Binding: a (var_no, object) pair 

145 # Bindings: List-of(Binding) 

146 # BindingsFactor: List-of(Bindings) 

147 # BindingsFactors: List-of(BindingsFactor) 

148 bindings_factors = [] 

149 for pos, cond in enumerate(self.conditions): 

150 if pos == cond_index: 

151 continue 

152 atoms = self.atoms_by_index[pos] 

153 assert atoms, "if we have no atoms, this should never be called" 

154 factor = [self._get_bindings(atom, cond) for atom in atoms] 

155 bindings_factors.append(factor) 

156 

157 eff_args = self.prepare_effect(new_atom, cond_index) 

158 

159 for bindings_list in itertools.product(*bindings_factors): 

160 bindings = itertools.chain(*bindings_list) 

161 for var_no, obj in bindings: 

162 eff_args[var_no] = obj 

163 enqueue_func(self.effect.predicate, eff_args)

166class ProjectRule(BuildRule): 

167 def __init__(self, effect, conditions): 

168 self.effect = effect 

169 self.conditions = conditions 

170 def validate(self): 

171 assert len(self.conditions) == 1 

172 def update_index(self, new_atom, cond_index): 

173 pass 

174 def fire(self, new_atom, cond_index, enqueue_func): 

175 effect_args = self.prepare_effect(new_atom, cond_index) 

176 enqueue_func(self.effect.predicate, effect_args)

178class Unifier: 

179 def __init__(self, rules): 

180 self.predicate_to_rule_generator = {} 

181 for rule in rules: 

182 for i, cond in enumerate(rule.conditions): 

183 self._insert_condition(rule, i) 

184 def unify(self, atom): 

185 result = [] 

186 generator = self.predicate_to_rule_generator.get(atom.predicate) 

187 if generator: 

188 generator.generate(atom, result) 

189 return result 

190 def _insert_condition(self, rule, cond_index): 

191 condition = rule.conditions[cond_index] 

192 root = self.predicate_to_rule_generator.get(condition.predicate) 

193 if not root: 

194 root = LeafGenerator() 

195 constant_arguments = [ 

196 (arg_index, arg) 

197 for (arg_index, arg) in enumerate(condition.args) 

198 if not isinstance(arg, int) and arg[0] != "?"] 

199 newroot = root._insert(constant_arguments, (rule, cond_index)) 

200 self.predicate_to_rule_generator[condition.predicate] = newroot 

201 def dump(self): 

202 predicates = sorted(self.predicate_to_rule_generator) 

203 print("Unifier:") 

204 for pred in predicates: 

205 print(" %s:" % pred) 

206 rule_gen = self.predicate_to_rule_generator[pred] 

207 rule_gen.dump(" " * 2)

209class LeafGenerator: 

210 index = sys.maxsize 

211 def __init__(self): 

212 self.matches = [] 

213 def empty(self): 

214 return not self.matches 

215 def generate(self, atom, result): 

216 result += self.matches 

217 def _insert(self, args, value): 

218 if not args: 

219 self.matches.append(value) 

220 return self 

221 else: 

222 root = LeafGenerator() 

223 root.matches.append(value) 

224 for arg_index, arg in args[::-1]: 

225 new_root = MatchGenerator(arg_index, LeafGenerator()) 

226 new_root.match_generator[arg] = root 

227 root = new_root 

228 root.matches = self.matches # can be swapped in C++ 

229 return root 

230 def dump(self, indent): 

231 for match in self.matches: 

232 print("%s%s" % (indent, match))

234class MatchGenerator: 

235 def __init__(self, index, next): 

236 self.index = index 

237 self.matches = [] 

238 self.match_generator = {} 

239 self.next = next 

240 def empty(self): 

241 return False 

242 def generate(self, atom, result): 

243 result += self.matches 

244 generator = self.match_generator.get(atom.args[self.index]) 

245 if generator: 

246 generator.generate(atom, result) 

247 self.next.generate(atom, result) 

248 def _insert(self, args, value): 

249 if not args: 

250 self.matches.append(value) 

251 return self 

252 else: 

253 arg_index, arg = args[0] 

254 if self.index < arg_index: 

255 self.next = self.next._insert(args, value) 

256 return self 

257 elif self.index > arg_index: 

258 new_parent = MatchGenerator(arg_index, self) 

259 new_branch = LeafGenerator()._insert(args[1:], value) 

260 new_parent.match_generator[arg] = new_branch 

261 return new_parent 

262 else: 

263 branch_generator = self.match_generator.get(arg) 

264 if not branch_generator: 

265 branch_generator = LeafGenerator() 

266 self.match_generator[arg] = branch_generator._insert( 

267 args[1:], value) 

268 return self 

269 def dump(self, indent): 

270 for match in self.matches: 

271 print("%s%s" % (indent, match)) 

272 for key in sorted(self.match_generator.keys()): 

273 print("%sargs[%s] == %s:" % (indent, self.index, key)) 

274 self.match_generator[key].dump(indent + " ") 

275 if not self.next.empty(): 

276 assert isinstance(self.next, MatchGenerator) 

277 print("%s[*]" % indent) 

278 self.next.dump(indent + " ")

280class Queue: 

281 def __init__(self, atoms): 

282 self.queue = atoms 

283 self.queue_pos = 0 

284 self.enqueued = set([(atom.predicate,) + tuple(atom.args) 

285 for atom in self.queue]) 

286 self.num_pushes = len(atoms) 

287 def __bool__(self): 

288 return self.queue_pos < len(self.queue) 

289 __nonzero__ = __bool__ 

290 def push(self, predicate, args): 

291 self.num_pushes += 1 

292 eff_tuple = (predicate,) + tuple(args) 

293 if eff_tuple not in self.enqueued: 

294 self.enqueued.add(eff_tuple) 

295 self.queue.append(pddl.Atom(predicate, list(args))) 

296 def pop(self): 

297 result = self.queue[self.queue_pos] 

298 self.queue_pos += 1 

299 return result

301def compute_model(prog): 

302 with timers.timing("Preparing model"): 

303 rules = convert_rules(prog) 

304 unifier = Unifier(rules) 

305 # unifier.dump() 

306 fact_atoms = sorted(fact.atom for fact in prog.facts) 

307 queue = Queue(fact_atoms) 

308 

309 print("Generated %d rules." % len(rules)) 

310 with timers.timing("Computing model"): 

311 relevant_atoms = 0 

312 auxiliary_atoms = 0 

313 while queue: 

314 next_atom = queue.pop() 

315 pred = next_atom.predicate 

316 if isinstance(pred, str) and "$" in pred: 

317 auxiliary_atoms += 1 

318 else: 

319 relevant_atoms += 1 

320 matches = unifier.unify(next_atom) 

321 for rule, cond_index in matches: 

322 rule.update_index(next_atom, cond_index) 

323 rule.fire(next_atom, cond_index, queue.push) 

324 print("%d relevant atoms" % relevant_atoms) 

325 print("%d auxiliary atoms" % auxiliary_atoms) 

326 print("%d final queue length" % len(queue.queue)) 

327 print("%d total queue pushes" % queue.num_pushes) 

328 return queue.queue
