4# Conditions (of any type) are immutable, because they need to 

5# be hashed occasionally. Immutability also allows more efficient comparison 

6# based on a precomputed hash value. 

7# 

8# Careful: Most other classes (e.g. Effects, Axioms, Actions) are not! 

9 

10class Condition(object): 

11 def __init__(self, parts): 

12 self.parts = tuple(parts) 

13 self.hash = hash((self.__class__, self.parts)) 

14 def __hash__(self): 

15 return self.hash 

16 def __ne__(self, other): 

17 return not self == other 

18 def __lt__(self, other): 

19 return self.hash < other.hash 

20 def __le__(self, other): 

21 return self.hash <= other.hash 

22 def dump(self, indent=" "): 

23 print("%s%s" % (indent, self._dump())) 

24 for part in self.parts: 

25 part.dump(indent + " ") 

26 def _dump(self): 

27 return self.__class__.__name__ 

28 def _postorder_visit(self, method_name, *args): 

29 part_results = [part._postorder_visit(method_name, *args) 

30 for part in self.parts] 

31 method = getattr(self, method_name, self._propagate) 

32 return method(part_results, *args) 

33 def _propagate(self, parts, *args): 

34 return self.change_parts(parts) 

35 def simplified(self): 

36 return self._postorder_visit("_simplified") 

37 def relaxed(self): 

38 return self._postorder_visit("_relaxed") 

39 def untyped(self): 

40 return self._postorder_visit("_untyped")

42 def uniquify_variables(self, type_map, renamings={}): 

43 # Cannot used _postorder_visit because this requires preorder 

44 # for quantified effects. 

45 if not self.parts: 

46 return self 

47 else: 

48 return self.__class__([part.uniquify_variables(type_map, renamings) 

49 for part in self.parts]) 

50 def to_untyped_strips(self): 

51 raise ValueError("Not a STRIPS condition: %s" % self.__class__.__name__) 

52 def instantiate(self, var_mapping, init_facts, fluent_facts, result): 

53 raise ValueError("Cannot instantiate condition: not normalized") 

54 def free_variables(self): 

55 result = set() 

56 for part in self.parts: 

57 result |= part.free_variables() 

58 return result

103class JunctorCondition(Condition): 

104 # Defining __eq__ blocks inheritance of __hash__, so must set it explicitly. 

105 __hash__ = Condition.__hash__ 

106 def __eq__(self, other): 

107 # Compare hash first for speed reasons. 

108 return (self.hash == other.hash and

114class Conjunction(JunctorCondition): 

115 def _simplified(self, parts): 

116 result_parts = [] 

117 for part in parts: 

118 if isinstance(part, Conjunction): 

119 result_parts += part.parts 

120 elif isinstance(part, Falsity): 

121 return Falsity() 

122 elif not isinstance(part, Truth): 

123 result_parts.append(part) 

124 if not result_parts: 

125 return Truth() 

126 if len(result_parts) == 1: 

127 return result_parts[0] 

128 return Conjunction(result_parts) 

129 def to_untyped_strips(self): 

130 result = [] 

131 for part in self.parts: 

132 result += part.to_untyped_strips() 

133 return result 

134 def instantiate(self, var_mapping, init_facts, fluent_facts, result): 

135 assert not result, "Condition not simplified" 

136 for part in self.parts: 

137 part.instantiate(var_mapping, init_facts, fluent_facts, result) 

138 def negate(self): 

139 return Disjunction([p.negate() for p in self.parts])

161class QuantifiedCondition(Condition):

163 __hash__ = Condition.__hash__ 

164 def __init__(self, parameters, parts): 

165 self.parameters = tuple(parameters) 

166 self.parts = tuple(parts) 

167 self.hash = hash((self.__class__, self.parameters, self.parts))

169 # Compare hash first for speed reasons.

172 self.parameters == other.parameters and 

173 self.parts == other.parts) 

174 def _dump(self, indent=" "):

221class Literal(Condition): 

222 # Defining __eq__ blocks inheritance of __hash__, so must set it explicitly. 

223 __hash__ = Condition.__hash__ 

224 parts = []

231 # Compare hash first for speed reasons. 

232 return (self.hash == other.hash and

235 self.args == other.args)
