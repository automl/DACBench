13DESCRIPTION = """Fast Downward driver script. 

14 

15Input files can be either a PDDL problem file (with an optional PDDL domain 

16file), in which case the driver runs both planner components (translate and 

17search), or a SAS+ translator output file, in which case the driver runs just 

18the search component. You can override this default behaviour by selecting 

19components manually with the flags below. The first component to be run 

20determines the required input files: 

21 

22--translate: [DOMAIN] PROBLEM 

23--search: TRANSLATE_OUTPUT 

24 

25Arguments given before the specified input files are interpreted by the driver 

26script ("driver options"). Arguments given after the input files are passed on 

27to the planner components ("component options"). In exceptional cases where no 

28input files are needed, use "--" to separate driver from component options. In 

29even more exceptional cases where input files begin with "--", use "--" to 

30separate driver options from input files and also to separate input files from 

31component options. 

32 

33By default, component options are passed to the search component. Use 

34"--translate-options" or "--search-options" within the component options to 

35override the default for the following options, until overridden again. (See 

36below for examples.)"""

117def _rindex(seq, element): 

118 """Like list.index, but gives the index of the *last* occurrence.""" 

119 seq = list(reversed(seq)) 

120 reversed_index = seq.index(element) 

121 return len(seq) - 1 - reversed_index

124def _split_off_filenames(planner_args): 

125 """Given the list of arguments to be passed on to the planner 

126 components, split it into a prefix of filenames and a suffix of 

127 options. Returns a pair (filenames, options). 

128 

129 If a "--" separator is present, the last such separator serves as 

130 the border between filenames and options. The separator itself is 

131 not returned. (This implies that "--" can be a filename, but never 

132 an option to a planner component.) 

133 

134 If no such separator is present, the first argument that begins 

135 with "-" and consists of at least two characters starts the list 

136 of options, and all previous arguments are filenames.""" 

137 

138 if "--" in planner_args: 

139 separator_pos = _rindex(planner_args, "--") 

140 num_filenames = separator_pos 

141 del planner_args[separator_pos] 

142 else: 

143 num_filenames = 0 

144 for arg in planner_args: 

145 # We treat "-" by itself as a filename because by common 

146 # convention it denotes stdin or stdout, and we might want 

147 # to support this later. 

148 if arg.startswith("-") and arg != "-": 

149 break 

150 num_filenames += 1 

151 return planner_args[:num_filenames], planner_args[num_filenames:]

154def _split_planner_args(parser, args): 

155 """Partition args.planner_args, the list of arguments for the 

156 planner components, into args.filenames, args.translate_options 

157 and args.search_options. Modifies args directly and removes the original 

158 args.planner_args list.""" 

159 

160 args.filenames, options = _split_off_filenames(args.planner_args) 

161 

162 args.translate_options = [] 

163 args.search_options = [] 

164 

165 curr_options = args.search_options 

166 for option in options: 

167 if option == "--translate-options": 

168 curr_options = args.translate_options 

169 elif option == "--search-options": 

170 curr_options = args.search_options 

171 else: 

172 curr_options.append(option)

175def _check_mutex_args(parser, args, required=False): 

176 for pos, (name1, is_specified1) in enumerate(args): 

177 for name2, is_specified2 in args[pos + 1:]: 

178 if is_specified1 and is_specified2: 

179 print_usage_and_exit_with_driver_input_error( 

180 parser, "cannot combine %s with %s" % (name1, name2)) 

181 if required and not any(is_specified for _, is_specified in args): 

182 print_usage_and_exit_with_driver_input_error( 

183 parser, "exactly one of {%s} has to be specified" % 

184 ", ".join(name for name, _ in args))

193def _set_components_automatically(parser, args): 

194 """Guess which planner components to run based on the specified 

195 filenames and set args.components accordingly. Currently 

196 implements some simple heuristics: 

197 

198 1. If there is exactly one input file and it looks like a 

199 Fast-Downward-generated file, run search only. 

200 2. Otherwise, run all components.""" 

201 

202 if len(args.filenames) == 1 and _looks_like_search_input(args.filenames[0]): 

203 args.components = ["search"] 

204 else: 

205 args.components = ["translate", "search"]

208def _set_components_and_inputs(parser, args): 

209 """Set args.components to the planner components to be run and set 

210 args.translate_inputs and args.search_input to the correct input 

211 filenames. 

212 

213 Rules: 

214 1. If any --run-xxx option is specified, then the union 

215 of the specified components is run. 

216 2. If nothing is specified, use automatic rules. See 

217 separate function.""" 

218 

219 args.components = [] 

220 if args.translate or args.run_all: 

221 args.components.append("translate") 

222 if args.search or args.run_all: 

223 args.components.append("search") 

224 

225 if not args.components: 

226 _set_components_automatically(parser, args) 

227 

228 # We implicitly activate validation in debug mode. However, for 

229 # validation we need the PDDL input files and a plan, therefore both 

230 # components must be active. 

231 if args.validate or (args.debug and len(args.components) == 2): 

232 args.components.append("validate") 

233 

234 args.translate_inputs = [] 

235 

236 assert args.components 

237 first = args.components[0] 

238 num_files = len(args.filenames) 

239 # When passing --help to any of the components (or -h to the 

240 # translator), we don't require input filenames and silently 

241 # swallow any that are provided. This is undocumented to avoid 

242 # cluttering the driver's --help output. 

243 if first == "translate": 

244 if "--help" in args.translate_options or "-h" in args.translate_options: 

245 args.translate_inputs = [] 

246 elif num_files == 1: 

247 task_file, = args.filenames 

248 domain_file = util.find_domain_filename(task_file) 

249 args.translate_inputs = [domain_file, task_file] 

250 elif num_files == 2: 

251 args.translate_inputs = args.filenames 

252 else: 

253 print_usage_and_exit_with_driver_input_error( 

254 parser, "translator needs one or two input files") 

255 elif first == "search": 

256 if "--help" in args.search_options: 

257 args.search_input = None 

258 elif num_files == 1: 

259 args.search_input, = args.filenames 

260 else: 

261 print_usage_and_exit_with_driver_input_error( 

262 parser, "search needs exactly one input file") 

263 else: 

264 assert False, first

277def _get_time_limit_in_seconds(limit, parser): 

278 match = re.match(r"^(\d+)(s|m|h)?$", limit, flags=re.I) 

279 if not match: 

280 print_usage_and_exit_with_driver_input_error(parser, "malformed time limit parameter: {}".format(limit)) 

281 time = int(match.group(1)) 

282 suffix = match.group(2) 

283 if suffix is not None: 

284 suffix = suffix.lower() 

285 if suffix == "m": 

286 time *= 60 

287 elif suffix == "h": 

288 time *= 3600 

289 return time

292def _get_memory_limit_in_bytes(limit, parser): 

293 match = re.match(r"^(\d+)(k|m|g)?$", limit, flags=re.I) 

294 if not match: 

295 print_usage_and_exit_with_driver_input_error(parser, "malformed memory limit parameter: {}".format(limit)) 

296 memory = int(match.group(1)) 

297 suffix = match.group(2) 

298 if suffix is not None: 

299 suffix = suffix.lower() 

300 if suffix == "k": 

301 memory *= 1024 

302 elif suffix is None or suffix == "m": 

303 memory *= 1024 * 1024 

304 elif suffix == "g": 

305 memory *= 1024 * 1024 * 1024 

306 return memory

449 

450 _check_mutex_args(parser, [

454

456 

457 _convert_limits_to_ints(parser, args)

459 if args.alias:

462 except KeyError: 

463 print_usage_and_exit_with_driver_input_error(

465 

466 if args.portfolio_bound is not None and not args.portfolio: 

467 print_usage_and_exit_with_driver_input_error(

473 print_usage_and_exit_with_driver_input_error( 

474 parser, "--portfolio-single_plan may only be used for portfolios.") 

475 

476 if not args.show_aliases and not args.cleanup: 

477 _set_components_and_inputs(parser, args) 

478 if "translate" not in args.components or "search" not in args.components: 

479 args.keep_sas_file = True 

480 

481 return args
