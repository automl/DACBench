18def invert_list(alist): 

19 result = defaultdict(list) 

20 for pos, arg in enumerate(alist): 

21 result[arg].append(pos) 

22 return result

31def find_unique_variables(action, invariant): 

32 # find unique names for invariant variables 

33 params = set([p.name for p in action.parameters]) 

34 for eff in action.effects: 

35 params.update([p.name for p in eff.parameters]) 

36 inv_vars = [] 

37 counter = itertools.count() 

38 for _ in range(invariant.arity()): 

39 while True: 

40 new_name = "?v%i" % next(counter) 

41 if new_name not in params: 

42 inv_vars.append(new_name) 

43 break 

44 return inv_vars

55def ensure_conjunction_sat(system, *parts): 

56 """Modifies the constraint system such that it is only solvable if the 

57 conjunction of all parts is satisfiable. 

58 

59 Each part must be an iterator, generator, or an iterable over 

60 literals.""" 

61 pos = defaultdict(set) 

62 neg = defaultdict(set) 

63 for literal in itertools.chain(*parts): 

64 if literal.predicate == "=": # use (in)equalities in conditions 

65 if literal.negated: 

66 n = constraints.NegativeClause([literal.args]) 

67 system.add_negative_clause(n) 

68 else: 

69 a = constraints.Assignment([literal.args]) 

70 system.add_assignment_disjunction([a]) 

71 else: 

72 if literal.negated: 

73 neg[literal.predicate].add(literal) 

74 else: 

75 pos[literal.predicate].add(literal) 

76 

77 for pred, posatoms in pos.items(): 

78 if pred in neg: 

79 for posatom in posatoms: 

80 for negatom in neg[pred]: 

81 parts = list(zip(negatom.args, posatom.args)) 

82 if parts: 

83 negative_clause = constraints.NegativeClause(parts) 

84 system.add_negative_clause(negative_clause)

87def ensure_cover(system, literal, invariant, inv_vars): 

88 """Modifies the constraint system such that it is only solvable if the 

89 invariant covers the literal""" 

90 a = invariant.get_covering_assignments(inv_vars, literal) 

91 assert(len(a) == 1) 

92 # if invariants could contain several parts of one predicate, this would 

93 # not be true but the depending code in parts relies on this assumption 

94 system.add_assignment_disjunction(a)

97def ensure_inequality(system, literal1, literal2): 

98 """Modifies the constraint system such that it is only solvable if the 

99 literal instantiations are not equal (ignoring whether one is negated and 

100 the other is not)""" 

101 if (literal1.predicate == literal2.predicate and 

102 literal1.args): 

103 parts = list(zip(literal1.args, literal2.args)) 

104 system.add_negative_clause(constraints.NegativeClause(parts))

107class InvariantPart: 

108 def __init__(self, predicate, order, omitted_pos=-1): 

109 self.predicate = predicate 

110 self.order = order 

111 self.omitted_pos = omitted_pos 

112 

113 def __eq__(self, other): 

114 # This implies equality of the omitted_pos component. 

115 return self.predicate == other.predicate and self.order == other.order 

116 

117 def __ne__(self, other): 

118 return self.predicate != other.predicate or self.order != other.order 

119 

120 def __le__(self, other): 

121 return self.predicate <= other.predicate or self.order <= other.order 

122 

123 def __lt__(self, other): 

124 return self.predicate < other.predicate or self.order < other.order 

125 

126 def __hash__(self): 

127 return hash((self.predicate, tuple(self.order))) 

128 

129 def __str__(self): 

130 var_string = " ".join(map(str, self.order)) 

131 omitted_string = "" 

132 if self.omitted_pos != -1: 

133 omitted_string = " [%d]" % self.omitted_pos 

134 return "%s %s%s" % (self.predicate, var_string, omitted_string) 

135 

136 def arity(self): 

137 return len(self.order) 

138 

139 def get_assignment(self, parameters, literal): 

140 equalities = [(arg, literal.args[argpos]) 

141 for arg, argpos in zip(parameters, self.order)] 

142 return constraints.Assignment(equalities) 

143 

144 def get_parameters(self, literal): 

145 return [literal.args[pos] for pos in self.order] 

146 

147 def instantiate(self, parameters): 

148 args = ["?X"] * (len(self.order) + (self.omitted_pos != -1)) 

149 for arg, argpos in zip(parameters, self.order): 

150 args[argpos] = arg 

151 return pddl.Atom(self.predicate, args)

153 def possible_mappings(self, own_literal, other_literal): 

154 allowed_omissions = len(other_literal.args) - len(self.order) 

155 if allowed_omissions not in (0, 1): 

156 return [] 

157 own_parameters = self.get_parameters(own_literal) 

158 arg_to_ordered_pos = invert_list(own_parameters) 

159 other_arg_to_pos = invert_list(other_literal.args) 

160 factored_mapping = [] 

161 

162 for key, other_positions in other_arg_to_pos.items(): 

163 own_positions = arg_to_ordered_pos.get(key, []) 

164 len_diff = len(own_positions) - len(other_positions) 

165 if len_diff >= 1 or len_diff <= -2 or len_diff == -1 and not allowed_omissions: 

166 return [] 

167 if len_diff: 

168 own_positions.append(-1) 

169 allowed_omissions = 0 

170 factored_mapping.append((other_positions, own_positions)) 

171 return instantiate_factored_mapping(factored_mapping) 

172 

173 def possible_matches(self, own_literal, other_literal): 

174 assert self.predicate == own_literal.predicate 

175 result = [] 

176 for mapping in self.possible_mappings(own_literal, other_literal): 

177 new_order = [None] * len(self.order) 

178 omitted = -1 

179 for (key, value) in mapping: 

180 if value == -1: 

181 omitted = key 

182 else: 

183 new_order[value] = key 

184 result.append(InvariantPart(other_literal.predicate, new_order, omitted)) 

185 return result

187 def matches(self, other, own_literal, other_literal): 

188 return self.get_parameters(own_literal) == other.get_parameters(other_literal)

191class Invariant: 

192 # An invariant is a logical expression of the type 

193 # forall V1...Vk: sum_(part in parts) weight(part, V1, ..., Vk) <= 1. 

194 # k is called the arity of the invariant. 

195 # A "part" is a symbolic fact only variable symbols in {V1, ..., Vk, X}; 

196 # the symbol X may occur at most once. 

197 

198 def __init__(self, parts): 

199 self.parts = frozenset(parts) 

200 self.predicates = set([part.predicate for part in parts]) 

201 self.predicate_to_part = dict([(part.predicate, part) for part in parts]) 

202 assert len(self.parts) == len(self.predicates) 

203 

204 def __eq__(self, other): 

205 return self.parts == other.parts 

206 

207 def __ne__(self, other): 

208 return self.parts != other.parts 

209 

210 def __lt__(self, other): 

211 return self.parts < other.parts 

212 

213 def __le__(self, other): 

214 return self.parts <= other.parts 

215 

216 def __hash__(self): 

217 return hash(self.parts) 

218 

219 def __str__(self): 

220 return "{%s}" % ", ".join(str(part) for part in self.parts) 

221 

222 def __repr__(self): 

223 return '<Invariant %s>' % self 

224 

225 def arity(self): 

226 return next(iter(self.parts)).arity() 

227 

228 def get_parameters(self, atom): 

229 return self.predicate_to_part[atom.predicate].get_parameters(atom) 

230 

231 def instantiate(self, parameters): 

232 return [part.instantiate(parameters) for part in self.parts] 

233 

234 def get_covering_assignments(self, parameters, atom): 

235 part = self.predicate_to_part[atom.predicate] 

236 return [part.get_assignment(parameters, atom)] 

237 # if there were more parts for the same predicate the list 

238 # contained more than one element

240 def check_balance(self, balance_checker, enqueue_func): 

241 # Check balance for this hypothesis. 

242 actions_to_check = set() 

243 for part in self.parts: 

244 actions_to_check |= balance_checker.get_threats(part.predicate) 

245 for action in actions_to_check: 

246 heavy_action = balance_checker.get_heavy_action(action) 

247 if self.operator_too_heavy(heavy_action): 

248 return False 

249 if self.operator_unbalanced(action, enqueue_func): 

250 return False 

251 return True

253 def operator_too_heavy(self, h_action): 

254 add_effects = [eff for eff in h_action.effects 

255 if not eff.literal.negated and 

256 self.predicate_to_part.get(eff.literal.predicate)] 

257 inv_vars = find_unique_variables(h_action, self) 

258 

259 if len(add_effects) <= 1: 

260 return False 

261 

262 for eff1, eff2 in itertools.combinations(add_effects, 2): 

263 system = constraints.ConstraintSystem() 

264 ensure_inequality(system, eff1.literal, eff2.literal) 

265 ensure_cover(system, eff1.literal, self, inv_vars) 

266 ensure_cover(system, eff2.literal, self, inv_vars) 

267 ensure_conjunction_sat(system, get_literals(h_action.precondition), 

268 get_literals(eff1.condition), 

269 get_literals(eff2.condition), 

270 [eff1.literal.negate()], 

271 [eff2.literal.negate()]) 

272 if system.is_solvable(): 

273 return True 

274 return False

276 def operator_unbalanced(self, action, enqueue_func): 

277 inv_vars = find_unique_variables(action, self) 

278 relevant_effs = [eff for eff in action.effects 

279 if self.predicate_to_part.get(eff.literal.predicate)] 

280 add_effects = [eff for eff in relevant_effs 

281 if not eff.literal.negated] 

282 del_effects = [eff for eff in relevant_effs 

283 if eff.literal.negated] 

284 for eff in add_effects: 

285 if self.add_effect_unbalanced(action, eff, del_effects, inv_vars, 

286 enqueue_func): 

287 return True 

288 return False

290 def minimal_covering_renamings(self, action, add_effect, inv_vars): 

291 """computes the minimal renamings of the action parameters such 

292 that the add effect is covered by the action. 

293 Each renaming is an constraint system""" 

294 

295 # add_effect must be covered 

296 assigs = self.get_covering_assignments(inv_vars, add_effect.literal) 

297 

298 # renaming of operator parameters must be minimal 

299 minimal_renamings = [] 

300 params = [p.name for p in action.parameters] 

301 for assignment in assigs: 

302 system = constraints.ConstraintSystem() 

303 system.add_assignment(assignment) 

304 mapping = assignment.get_mapping() 

305 if len(params) > 1: 

306 for (n1, n2) in itertools.combinations(params, 2): 

307 if mapping.get(n1, n1) != mapping.get(n2, n2): 

308 negative_clause = constraints.NegativeClause([(n1, n2)]) 

309 system.add_negative_clause(negative_clause) 

310 minimal_renamings.append(system) 

311 return minimal_renamings

313 def add_effect_unbalanced(self, action, add_effect, del_effects, 

314 inv_vars, enqueue_func): 

315 

316 minimal_renamings = self.minimal_covering_renamings(action, add_effect, 

317 inv_vars) 

318 

319 lhs_by_pred = defaultdict(list) 

320 for lit in itertools.chain(get_literals(action.precondition), 

321 get_literals(add_effect.condition), 

322 get_literals(add_effect.literal.negate())): 

323 lhs_by_pred[lit.predicate].append(lit) 

324 

325 for del_effect in del_effects: 

326 minimal_renamings = self.unbalanced_renamings(del_effect, add_effect, 

327 inv_vars, lhs_by_pred, minimal_renamings) 

328 if not minimal_renamings: 

329 return False 

330 

331 # Otherwise, the balance check fails => Generate new candidates. 

332 self.refine_candidate(add_effect, action, enqueue_func) 

333 return True

335 def refine_candidate(self, add_effect, action, enqueue_func): 

336 """refines the candidate for an add effect that is unbalanced in the 

337 action and adds the refined one to the queue""" 

338 part = self.predicate_to_part[add_effect.literal.predicate] 

339 for del_eff in [eff for eff in action.effects if eff.literal.negated]: 

340 if del_eff.literal.predicate not in self.predicate_to_part: 

341 for match in part.possible_matches(add_effect.literal, 

342 del_eff.literal): 

343 enqueue_func(Invariant(self.parts.union((match,))))

345 def unbalanced_renamings(self, del_effect, add_effect, 

346 inv_vars, lhs_by_pred, unbalanced_renamings): 

347 """returns the renamings from unbalanced renamings for which 

348 the del_effect does not balance the add_effect.""" 

349 

350 system = constraints.ConstraintSystem() 

351 ensure_cover(system, del_effect.literal, self, inv_vars) 

352 

353 # Since we may only rename the quantified variables of the delete effect 

354 # we need to check that "renamings" of constants are already implied by 

355 # the unbalanced_renaming (of the of the operator parameters). The 

356 # following system is used as a helper for this. It builds a conjunction 

357 # that formulates that the constants are NOT renamed accordingly. We 

358 # below check that this is impossible with each unbalanced renaming. 

359 check_constants = False 

360 constant_test_system = constraints.ConstraintSystem() 

361 for a,b in system.combinatorial_assignments[0][0].equalities: 

362 # first 0 because the system was empty before we called ensure_cover 

363 # second 0 because ensure_cover only adds assignments with one entry 

364 if b[0] != "?": 

365 check_constants = True 

366 neg_clause = constraints.NegativeClause([(a,b)]) 

367 constant_test_system.add_negative_clause(neg_clause) 

368 

369 ensure_inequality(system, add_effect.literal, del_effect.literal) 

370 

371 still_unbalanced = [] 

372 for renaming in unbalanced_renamings: 

373 if check_constants: 

374 new_sys = constant_test_system.combine(renaming) 

375 if new_sys.is_solvable(): 

376 # it is possible that the operator arguments are not 

377 # mapped to constants as required for covering the delete 

378 # effect 

379 still_unbalanced.append(renaming) 

380 continue 

381 

382 new_sys = system.combine(renaming) 

383 if self.lhs_satisfiable(renaming, lhs_by_pred): 

384 implies_system = self.imply_del_effect(del_effect, lhs_by_pred) 

385 if not implies_system: 

386 still_unbalanced.append(renaming) 

387 continue 

388 new_sys = new_sys.combine(implies_system) 

389 if not new_sys.is_solvable(): 

390 still_unbalanced.append(renaming) 

391 return still_unbalanced

393 def lhs_satisfiable(self, renaming, lhs_by_pred): 

394 system = renaming.copy() 

395 ensure_conjunction_sat(system, *itertools.chain(lhs_by_pred.values())) 

396 return system.is_solvable()

398 def imply_del_effect(self, del_effect, lhs_by_pred): 

399 """returns a constraint system that is solvable if lhs implies 

400 the del effect (only if lhs is satisfiable). If a solvable 

401 lhs never implies the del effect, return None.""" 

402 # del_effect.cond and del_effect.atom must be implied by lhs 

403 implies_system = constraints.ConstraintSystem() 

404 for literal in itertools.chain(get_literals(del_effect.condition), 

405 [del_effect.literal.negate()]): 

406 poss_assignments = [] 

407 for match in lhs_by_pred[literal.predicate]: 

408 if match.negated != literal.negated: 

409 continue 

410 else: 

411 a = constraints.Assignment(list(zip(literal.args, match.args))) 

412 poss_assignments.append(a) 

413 if not poss_assignments: 

414 return None 

415 implies_system.add_assignment_disjunction(poss_assignments) 

416 return implies_system
