3"""Tarjan's algorithm for maximal strongly connected components. 

4 

5We provide two versions of the algorithm for different graph 

6representations. 

7 

8Since the original recursive version exceeds python's maximal 

9recursion depth on some planning instances, this is an iterative 

10version with an explicit recursion stack (iter_stack). 

11 

12Note that the derived graph where each SCC is a single "supernode" is 

13necessarily acyclic. The SCCs returned by the algorithm are in a 

14topological sort order with respect to this derived DAG. 

15""" 

16 

17from collections import defaultdict 

18 

19__all__ = ["get_sccs_adjacency_list", "get_sccs_adjacency_dict"] 

20 

21 

22def get_sccs_adjacency_list(adjacency_list): 

23 """Compute SCCs for a graph represented as an adjacency list. 

24 

25 `adjacency_list` is a list (or similar data structure) whose 

26 indices correspond to the graph nodes. For example, if 

27 `len(adjacency_list)` is N, the graph nodes are {0, ..., N-1}. 

28 

29 For every node `u`, `adjacency_list[u]` is the list (or similar data 

30 structure) of successors of `u`. 

31 

32 Returns a list of lists that defines a partition of {0, ..., N-1}, 

33 where each block in the partition is an SCC of the graph, and 

34 the partition is given in a topologically sort order.""" 

35 return StronglyConnectedComponentComputation(adjacency_list).get_result() 

36 

37def get_sccs_adjacency_dict(adjacency_dict): 

38 """Compute SCCs for a graph represented as an adjacency dict. 

39 

40 `adjacency_dict` is a dictionary whose keys are the vertices of 

41 the graph. 

42 

43 For every node `u`, adjacency_dict[u]` is the list (or similar 

44 data structure) of successors of `u`. 

45 

46 Returns a list of lists that defines a partition of the graph 

47 nodes, where each block in the partition is an SCC of the graph, 

48 and the partition is given in a topologically sort order.""" 

49 node_to_index = {} 

50 index_to_node = [] 

51 for index, node in enumerate(adjacency_dict): 

52 node_to_index[node] = index 

53 index_to_node.append(node) 

54 

55 adjacency_list = [] 

56 for index, node in enumerate(index_to_node): 

57 successors = adjacency_dict[node] 

58 successor_indices = [node_to_index[v] for v in successors] 

59 adjacency_list.append(successor_indices) 

60 

61 result_indices = get_sccs_adjacency_list(adjacency_list) 

62 

63 result = [] 

64 for block_indices in result_indices: 

65 block = [index_to_node[index] for index in block_indices] 

66 result.append(block) 

67 return result

70class StronglyConnectedComponentComputation(object): 

71 def __init__(self, unweighted_graph): 

72 self.graph = unweighted_graph 

73 self.BEGIN, self.CONTINUE, self.RETURN = 0, 1, 2 # "recursion" handling 

74 

75 def get_result(self): 

76 self.indices = dict() 

77 self.lowlinks = defaultdict(lambda: -1) 

78 self.stack_indices = dict() 

79 self.current_index = 0 

80 self.stack = [] 

81 self.sccs = [] 

82 

83 for i in range(len(self.graph)): 

84 if i not in self.indices: 

85 self.visit(i) 

86 self.sccs.reverse() 

87 return self.sccs 

88 

89 def visit(self, vertex): 

90 iter_stack = [(vertex, None, None, self.BEGIN)] 

91 while iter_stack: 

92 v, w, succ_index, state = iter_stack.pop() 

93 

94 if state == self.BEGIN: 

95 self.current_index += 1 

96 self.indices[v] = self.current_index 

97 self.lowlinks[v] = self.current_index 

98 self.stack_indices[v] = len(self.stack) 

99 self.stack.append(v) 

100 

101 iter_stack.append((v, None, 0, self.CONTINUE)) 

102 elif state == self.CONTINUE: 

103 successors = self.graph[v] 

104 if succ_index == len(successors): 

105 if self.lowlinks[v] == self.indices[v]: 

106 stack_index = self.stack_indices[v] 

107 scc = self.stack[stack_index:] 

108 del self.stack[stack_index:] 

109 for n in scc: 

110 del self.stack_indices[n] 

111 self.sccs.append(scc) 

112 else: 

113 w = successors[succ_index] 

114 if w not in self.indices: 

115 iter_stack.append((v, w, succ_index, self.RETURN)) 

116 iter_stack.append((w, None, None, self.BEGIN)) 

117 else: 

118 if w in self.stack_indices: 

119 self.lowlinks[v] = min(self.lowlinks[v], 

120 self.indices[w]) 

121 iter_stack.append( 

122 (v, None, succ_index + 1, self.CONTINUE)) 

123 elif state == self.RETURN: 

124 self.lowlinks[v] = min(self.lowlinks[v], self.lowlinks[w]) 

125 iter_stack.append((v, None, succ_index + 1, self.CONTINUE))
