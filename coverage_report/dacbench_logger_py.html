71def flatten_log_entry(log_entry: Dict) -> List[Dict]: 

72 """ 

73 Transforms a log entry of format like 

74 

75 

76 { 

77 'step': 0, 

78 'episode': 2, 

79 'some_value': { 

80 'values' : [34, 45], 

81 'times':['28-12-20 16:20:53', '28-12-20 16:21:30'], 

82 } 

83 } 

84 into 

85 [ 

86 { 'step': 0,'episode': 2, 'value': 34, 'time': '28-12-20 16:20:53'}, 

87 { 'step': 0,'episode': 2, 'value': 45, 'time': '28-12-20 16:21:30'} 

88 ] 

89 

90 Parameters 

91 ---------- 

92 log_entry: Dict 

93 A log entry 

94 

95 Returns 

96 ------- 

97 """ 

98 dict_entries, top_level_entries = split( 

99 lambda item: isinstance(item[1], dict), log_entry.items() 

100 ) 

101 rows = [] 

102 for value_name, value_dict in dict_entries: 

103 current_rows = ( 

104 dict( 

105 top_level_entries 

106 + [("value", value), ("time", time), ("name", value_name)] 

107 ) 

108 for value, time in zip(value_dict["values"], value_dict["times"]) 

109 ) 

110 

111 rows.extend(map(dict, current_rows)) 

112 

113 return rows

133def log2dataframe( 

134 logs: List[dict], wide: bool = False, drop_columns: List[str] = ["time"] 

135) -> pd.DataFrame: 

136 """ 

137 Converts a list of log entries to a pandas dataframe. 

138 

139 Usually used in combination with load_dataframe. 

140 

141 Parameters 

142 ---------- 

143 logs: List 

144 List of log entries 

145 wide: bool 

146 wide=False (default) produces a dataframe with columns (episode, step, time, name, value) 

147 wide=True returns a dataframe (episode, step, time, name_1, name_2, ...) if the variable name_n has not been logged 

148 at (episode, step, time) name_n is NaN. 

149 drop_columns: List[str] 

150 List of column names to be dropped (before reshaping the long dataframe) mostly used in combination 

151 with wide=True to reduce NaN values 

152 

153 Returns 

154 ------- 

155 dataframe 

156 """ 

157 

158 flat_logs = map(flatten_log_entry, logs) 

159 rows = reduce(lambda l1, l2: l1 + l2, flat_logs) 

160 

161 dataframe = pd.DataFrame(rows) 

162 dataframe.time = pd.to_datetime(dataframe.time) 

163 

164 if drop_columns is not None: 

165 dataframe = dataframe.drop(columns=drop_columns) 

166 

167 dataframe = dataframe.infer_objects() 

168 list_column_candidates = dataframe.dtypes == object 

169 

170 for i, candidate in enumerate(list_column_candidates): 

171 if candidate: 

172 dataframe.iloc[:, i] = dataframe.iloc[:, i].apply( 

173 lambda x: list_to_tuple(x) if isinstance(x, list) else x 

174 ) 

175 

176 if wide: 

177 primary_index_columns = ["episode", "step"] 

178 field_id_column = "name" 

179 additional_columns = list( 

180 set(dataframe.columns) 

181 - set(primary_index_columns + ["time", "value", field_id_column]) 

182 ) 

183 index_columns = primary_index_columns + additional_columns + [field_id_column] 

184 dataframe = dataframe.set_index(index_columns) 

185 dataframe = dataframe.unstack() 

186 dataframe.reset_index(inplace=True) 

187 dataframe.columns = [a if b == "" else b for a, b in dataframe.columns] 

188 

189 return dataframe.infer_objects()

631 def __space_dict(key: str, value, space_info): 

632 if isinstance(value, np.ndarray) and len(value.shape) == 0: 

633 value = value.item() 

634 

635 if isinstance(value, Number): 

636 if space_info is None: 

637 data = {key: value}

656 elif isinstance(value, dict):
