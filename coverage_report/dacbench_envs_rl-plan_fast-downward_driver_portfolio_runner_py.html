5""" Module for running planner portfolios. 

6 

7Memory limits: We apply the same memory limit that is given to the 

8plan script to each planner call. Note that this setup does not work if 

9the sum of the memory usage of the Python process and the planner calls 

10is limited. In this case the Python process might get killed although 

11we would like to kill only the single planner call and continue with 

12the remaining configurations. If we ever want to support this scenario 

13we will have to reduce the memory limit of the planner calls by the 

14amount of memory that the Python process needs. On maia for example 

15this amounts to 128MB of reserved virtual memory. We can make Python 

16reserve less space by lowering the soft limit for virtual memory before 

17the process is started. 

18"""

34def adapt_heuristic_cost_type(arg, cost_type): 

35 if cost_type == "normal": 

36 transform = "no_transform()" 

37 else: 

38 transform = "adapt_costs({})".format(cost_type) 

39 return arg.replace("H_COST_TRANSFORM", transform)

42def adapt_args(args, search_cost_type, heuristic_cost_type, plan_manager): 

43 g_bound = plan_manager.get_next_portfolio_cost_bound() 

44 plan_counter = plan_manager.get_plan_counter() 

45 print("g bound: %s" % g_bound) 

46 print("next plan number: %d" % (plan_counter + 1)) 

47 

48 for index, arg in enumerate(args): 

49 if arg == "--evaluator" or arg == "--heuristic": 

50 heuristic = args[index + 1] 

51 heuristic = adapt_heuristic_cost_type(heuristic, heuristic_cost_type) 

52 args[index + 1] = heuristic 

53 elif arg == "--search": 

54 search = args[index + 1] 

55 if "bound=BOUND" not in search: 

56 returncodes.exit_with_driver_critical_error( 

57 "Satisficing portfolios need the string " 

58 "\"bound=BOUND\" in each search configuration. " 

59 "See the FDSS portfolios for examples.") 

60 for name, value in [ 

61 ("BOUND", g_bound), 

62 ("S_COST_TYPE", search_cost_type)]: 

63 search = search.replace(name, str(value)) 

64 search = adapt_heuristic_cost_type(search, heuristic_cost_type) 

65 args[index + 1] = search 

66 break

69def run_search(executable, args, sas_file, plan_manager, time, memory): 

70 complete_args = [executable] + args + [ 

71 "--internal-plan-file", plan_manager.get_plan_prefix()] 

72 print("args: %s" % complete_args) 

73 

74 try: 

75 exitcode = call.check_call( 

76 "search", complete_args, stdin=sas_file, 

77 time_limit=time, memory_limit=memory) 

78 except subprocess.CalledProcessError as err: 

79 exitcode = err.returncode 

80 print("exitcode: %d" % exitcode) 

81 print() 

82 return exitcode

85def compute_run_time(timeout, configs, pos): 

86 remaining_time = timeout - util.get_elapsed_time() 

87 print("remaining time: {}".format(remaining_time)) 

88 relative_time = configs[pos][0] 

89 remaining_relative_time = sum(config[0] for config in configs[pos:]) 

90 print("config {}: relative time {}, remaining {}".format( 

91 pos, relative_time, remaining_relative_time)) 

92 # For the last config we have relative_time == remaining_relative_time, so 

93 # we use all of the remaining time at the end. 

94 return remaining_time * relative_time / remaining_relative_time

97def run_sat_config(configs, pos, search_cost_type, heuristic_cost_type, 

98 executable, sas_file, plan_manager, timeout, memory): 

99 run_time = compute_run_time(timeout, configs, pos) 

100 if run_time <= 0: 

101 return None 

102 _, args_template = configs[pos] 

103 args = list(args_template) 

104 adapt_args(args, search_cost_type, heuristic_cost_type, plan_manager) 

105 if not plan_manager.abort_portfolio_after_first_plan(): 

106 args.extend([ 

107 "--internal-previous-portfolio-plans", 

108 str(plan_manager.get_plan_counter())]) 

109 result = run_search(executable, args, sas_file, plan_manager, run_time, memory) 

110 plan_manager.process_new_plans() 

111 return result

114def run_sat(configs, executable, sas_file, plan_manager, final_config, 

115 final_config_builder, timeout, memory): 

116 # If the configuration contains S_COST_TYPE or H_COST_TRANSFORM and the task 

117 # has non-unit costs, we start by treating all costs as one. When we find 

118 # a solution, we rerun the successful config with real costs. 

119 heuristic_cost_type = "one" 

120 search_cost_type = "one" 

121 changed_cost_types = False 

122 while configs: 

123 configs_next_round = [] 

124 for pos, (relative_time, args) in enumerate(configs): 

125 exitcode = run_sat_config( 

126 configs, pos, search_cost_type, heuristic_cost_type, 

127 executable, sas_file, plan_manager, timeout, memory) 

128 if exitcode is None: 

129 return 

130 

131 yield exitcode 

132 if exitcode == returncodes.SEARCH_UNSOLVABLE: 

133 return 

134 

135 if exitcode == returncodes.SUCCESS: 

136 if plan_manager.abort_portfolio_after_first_plan(): 

137 return 

138 configs_next_round.append((relative_time, args)) 

139 if (not changed_cost_types and can_change_cost_type(args) and 

140 plan_manager.get_problem_type() == "general cost"): 

141 print("Switch to real costs and repeat last run.") 

142 changed_cost_types = True 

143 search_cost_type = "normal" 

144 heuristic_cost_type = "plusone" 

145 exitcode = run_sat_config( 

146 configs, pos, search_cost_type, heuristic_cost_type, 

147 executable, sas_file, plan_manager, timeout, memory) 

148 if exitcode is None: 

149 return 

150 

151 yield exitcode 

152 if exitcode == returncodes.SEARCH_UNSOLVABLE: 

153 return 

154 if final_config_builder: 

155 print("Build final config.") 

156 final_config = final_config_builder(args) 

157 break 

158 

159 if final_config: 

160 break 

161 

162 # Only run the successful configs in the next round. 

163 configs = configs_next_round 

164 

165 if final_config: 

166 print("Abort portfolio and run final config.") 

167 exitcode = run_sat_config( 

168 [(1, final_config)], 0, search_cost_type, 

169 heuristic_cost_type, executable, sas_file, plan_manager, 

170 timeout, memory) 

171 if exitcode is not None: 

172 yield exitcode

175def run_opt(configs, executable, sas_file, plan_manager, timeout, memory): 

176 for pos, (relative_time, args) in enumerate(configs): 

177 run_time = compute_run_time(timeout, configs, pos) 

178 exitcode = run_search(executable, args, sas_file, plan_manager, 

179 run_time, memory) 

180 yield exitcode 

181 

182 if exitcode in [returncodes.SUCCESS, returncodes.SEARCH_UNSOLVABLE]: 

183 break

190def get_portfolio_attributes(portfolio): 

191 attributes = {} 

192 with open(portfolio) as portfolio_file: 

193 content = portfolio_file.read() 

194 try: 

195 exec(content, attributes) 

196 except Exception: 

197 returncodes.exit_with_driver_critical_error( 

198 "The portfolio %s could not be loaded. Maybe it still " 

199 "uses the old portfolio syntax? See the FDSS portfolios " 

200 "for examples using the new syntax." % portfolio) 

201 if "CONFIGS" not in attributes: 

202 returncodes.exit_with_driver_critical_error("portfolios must define CONFIGS") 

203 if "OPTIMAL" not in attributes: 

204 returncodes.exit_with_driver_critical_error("portfolios must define OPTIMAL") 

205 return attributes

208def run(portfolio, executable, sas_file, plan_manager, time, memory): 

209 """ 

210 Run the configs in the given portfolio file. 

211 

212 The portfolio is allowed to run for at most *time* seconds and may 

213 use a maximum of *memory* bytes. 

214 """ 

215 attributes = get_portfolio_attributes(portfolio) 

216 configs = attributes["CONFIGS"] 

217 optimal = attributes["OPTIMAL"] 

218 final_config = attributes.get("FINAL_CONFIG") 

219 final_config_builder = attributes.get("FINAL_CONFIG_BUILDER") 

220 if "TIMEOUT" in attributes: 

221 returncodes.exit_with_driver_input_error( 

222 "The TIMEOUT attribute in portfolios has been removed. " 

223 "Please pass a time limit to fast-downward.py.") 

224 

225 if time is None: 

226 if os.name == "nt": 

227 returncodes.exit_with_driver_unsupported_error(limits.RESOURCE_MODULE_MISSING_MSG) 

228 else: 

229 returncodes.exit_with_driver_input_error( 

230 "Portfolios need a time limit. Please pass --search-time-limit " 

231 "or --overall-time-limit to fast-downward.py.") 

232 

233 timeout = util.get_elapsed_time() + time 

234 

235 if optimal: 

236 exitcodes = run_opt( 

237 configs, executable, sas_file, plan_manager, timeout, memory) 

238 else: 

239 exitcodes = run_sat( 

240 configs, executable, sas_file, plan_manager, final_config, 

241 final_config_builder, timeout, memory) 

242 return returncodes.generate_portfolio_exitcode(exitcodes)
